\section{Introducción}
\begin{frame}{Introducción}
  \begin{itemize}
    \item La \alert{balanza de pagos} registra todas las transacciones de una economía con el resto del mundo y refleja la solidez de su posición externa.
    \item El Perú es una economía \textbf{pequeña, abierta y dependiente de materias primas}: su sector externo está atado a los precios internacionales de los metales.
    \item \textbf{Objetivo:} describir la estructura y la evolución de las cuentas externas peruanas entre \textbf{2016 y 2025}, con datos oficiales del BCRP.
    \item Muchos movimientos responden a eventos concretos: COVID-19, guerra Rusia--Ucrania, El Niño y el ciclo de precios de los metales.
  \end{itemize}
\end{frame}

\section{Marco Conceptual}
\begin{frame}{¿Qué es la balanza de pagos?}
  \begin{itemize}
    \item Registro contable y sistemático de las transacciones entre \textbf{residentes y no residentes} en un período.
    \item Se rige por el principio de \textbf{partida doble} (cada operación genera un crédito y un débito) y el criterio del \textbf{devengado}.
    \item En el Perú la elabora el \textbf{BCRP}, siguiendo el Manual del FMI (MBPI6) para asegurar comparabilidad internacional.
  \end{itemize}
  \vfill
  \begin{block}{Tres grandes bloques}
    Cuenta Corriente \;$\rightarrow$\; Cuenta Financiera \;$\rightarrow$\; Reservas Internacionales Netas (RIN)
  \end{block}
\end{frame}

\begin{frame}{Componentes de las cuentas externas}
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \textbf{Cuenta Corriente (flujos reales)}
      \begin{itemize}\small
        \item Balanza comercial de bienes (FOB)
        \item Balanza de servicios
        \item Ingreso primario (renta de factores)
        \item Ingreso secundario (remesas)
      \end{itemize}
      \vspace{0.5em}
      \textbf{Cuenta Financiera (activos/pasivos)}
      \begin{itemize}\small
        \item Sector privado (IDE, cartera, créditos)
        \item Sector público (bonos soberanos)
        \item Capitales de corto plazo
      \end{itemize}
    \end{column}
    \begin{column}{0.5\textwidth}
      \textbf{Reservas Internacionales Netas}
      \begin{itemize}\small
        \item Colchón financiero ante choques externos
        \item Capacidad de intervención cambiaria
        \item Respaldo de solvencia (menor riesgo país)
      \end{itemize}
      \vspace{0.8em}
      \begin{alertblock}{Convención de signo (MBPI6)}
        \footnotesize La Cuenta Financiera se mide como \emph{Activos $-$ Pasivos}: un valor \textbf{negativo} = \textbf{entrada} neta de capitales; positivo = salida/acumulación de activos.
      \end{alertblock}
    \end{column}
  \end{columns}
\end{frame}

\section{Identidades Contables}
\begin{frame}{Identidades fundamentales}
  \textbf{1. Resultado de la balanza de pagos} (equivale a la variación de RIN por flujos):
  \begin{equation*}
    CC - CF + E\&O + \text{Fin.Exc.} = \Delta RIN
  \end{equation*}
  {\footnotesize $CC$: cuenta corriente; $CF$: cuenta financiera; $E\&O$: errores y omisiones; Fin.Exc.: financiamiento excepcional.}

  \vspace{0.6em}
  \textbf{2. Desagregación de la cuenta corriente:}
  \begin{equation*}
    CC = BC_{\text{bienes}} + BC_{\text{servicios}} + IP + IS
  \end{equation*}
  {\footnotesize $BC$: balanza comercial; $IP$: ingreso primario; $IS$: ingreso secundario.}

  \vspace{0.6em}
  \textbf{3. Enfoque ahorro--inversión:}
  \begin{equation*}
    CC = S - I
  \end{equation*}
  {\footnotesize $S$: ahorro nacional; $I$: inversión interna. Un superávit sostenido implica $S > I$.}
\end{frame}

\begin{frame}{Cuenta corriente: composición del saldo}
  \small
  \begin{block}{Composición}
    El saldo de la \textbf{balanza comercial de bienes} (exportaciones $-$ importaciones) frente al saldo de \textbf{servicios}, año a año.
  \end{block}
  \begin{itemize}
    \item \textbf{Bienes (positivo y creciente):} el Perú vende más de lo que compra en mercancías; la brecha se dispara desde 2021 por los altos precios de metales.
    \item \textbf{Servicios (siempre negativo):} pagamos más fletes, seguros y turismo al exterior de lo que cobramos $\Rightarrow$ déficit estructural.
    \item El superávit de bienes supera ampliamente al déficit de servicios, y ese diferencial determina el paso de la cuenta corriente a superávit.
  \end{itemize}
\end{frame}

\begin{frame}{Cuenta corriente: variación interanual}
  \small
  \begin{itemize}
    \item \textbf{2019 $\to$ 2020 (mejora):} no fue por vender más, sino por \emph{comprar menos}. El confinamiento frenó la inversión y el consumo importado.
    \item \textbf{2020 $\to$ 2022 (deterioro):} al reactivarse, las importaciones se dispararon y los fletes encarecieron el valor CIF de lo importado; el saldo se volvió muy negativo.
    \item \textbf{2022 $\to$ 2023 (giro):} el precio del cobre/oro elevó el valor exportado más rápido que las importaciones $\Rightarrow$ la brecha se cierra y pasa a superávit.
    \item \textbf{2023 $\to$ 2025 (aceleración):} volumen minero + precios altos sostenidos; la balanza de bienes crece más rápido que el déficit de servicios e ingreso primario.
  \end{itemize}
  \centering\footnotesize\itshape Un saldo se fortalece cuando las exportaciones (precio $\times$ volumen) crecen por encima de las importaciones.
\end{frame}

\begin{frame}{Ingreso primario vs. secundario: la diferencia clave}
  \small
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \textbf{Ingreso primario (renta)} \; {\color{acc}$\downarrow$ resta}
      \begin{itemize}\footnotesize
        \item Utilidades y dividendos que las mineras extranjeras \textbf{envían a sus matrices}, más intereses de deuda.
        \item \textbf{Procíclico:} si sube el precio del cobre, suben las exportaciones \emph{pero también} las utilidades que salen del país.
        \item Por eso crece en paralelo al superávit comercial y lo modera.
      \end{itemize}
    \end{column}
    \begin{column}{0.5\textwidth}
      \textbf{Ingreso secundario (transferencias)} \; {\color{acc}$\uparrow$ suma}
      \begin{itemize}\footnotesize
        \item Sobre todo \textbf{remesas} de peruanos en el exterior: entran sin contraparte.
        \item Flujo \textbf{estable y resiliente}, incluso en crisis $\Rightarrow$ actúa como amortiguador.
      \end{itemize}
    \end{column}
  \end{columns}
  \vspace{0.6em}
  \centering\footnotesize\itshape El auge de precios que expande las exportaciones también incrementa la salida de utilidades mineras.
\end{frame}

\begin{frame}{Ingreso primario: profundización del déficit}
  \small
  \begin{itemize}
    \item El déficit pasa de $-9\,083$ (2016) a $-23\,104$ (2025): casi se triplica, pero \textbf{no es una señal negativa por sí sola}.
    \item Crece \emph{porque} las mineras ganan más: a mayor precio del cobre/oro, mayores utilidades $\Rightarrow$ mayor repatriación de dividendos.
    \item Es la \textbf{contracara} del auge exportador: el mismo shock de precios que infla las exportaciones infla la salida de rentas.
    \item Debe leerse neto contra la balanza de bienes; en ese neto, el país gana.
  \end{itemize}
  \centering\footnotesize\itshape Un mayor déficit refleja mayor rentabilidad minera, no un deterioro de la posición externa.
\end{frame}

\begin{frame}{Cuenta financiera: financiamiento externo}
  \small
  \begin{block}{Convención de signo (MBPI6)}
    Signo \textbf{negativo} $\Rightarrow$ entrada de capitales (financiamiento externo). Signo \textbf{positivo} $\Rightarrow$ salida de capitales o acumulación de activos externos.
  \end{block}
  \begin{itemize}\footnotesize
    \item Mientras la cuenta corriente fue \textbf{deficitaria} (2016--2022), la financiera fue \textbf{negativa}: el déficit se cubría con inversión extranjera, bonos y créditos.
    \item Al pasar la corriente a \textbf{superávit} (2023), el país deja de necesitar financiamiento y empieza a \textbf{colocar recursos afuera}.
    \item \textbf{2021 (pico de $-15\,385$):} fuerte entrada de capitales en plena reactivación pospandemia, para financiar el rebote de importaciones.
    \item Por identidad contable, un superávit en cuenta corriente implica que el país presta al resto del mundo.
  \end{itemize}
\end{frame}

\begin{frame}{Errores y Omisiones Netos}
  \small
  \begin{itemize}
    \item Es una \textbf{partida de ajuste}: como la BP se arma con miles de fuentes distintas, los registros nunca calzan perfecto.
    \item Recoge esa diferencia estadística para que la identidad contable \textbf{cierre} (cuadre a cero).
    \item \textbf{No es una cuenta económica real}, sino un residuo; por eso su signo cambia todos los años (de $+911$ en 2019 a $-7\,492$ en 2025).
    \item Un valor muy grande sugiere flujos no capturados bien (p. ej. movimientos de capital difíciles de medir), por lo que se vigila pero no se interpreta como tendencia.
  \end{itemize}
\end{frame}

\begin{frame}{Reservas Internacionales Netas: variación anual}
  \small
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \textbf{Años que \alert{suben} (2019, 2024, 2025)}
      \begin{itemize}\footnotesize
        \item Resultado de BP positivo (entran más divisas de las que salen).
        \item En 2025 se suma el \textbf{efecto valuación}: el oro que ya tenía el BCRP vale más, sin que entre un solo dólar nuevo.
      \end{itemize}
    \end{column}
    \begin{column}{0.5\textwidth}
      \textbf{Años que \alert{bajan} (2018, 2022, 2023)}
      \begin{itemize}\footnotesize
        \item Resultado de BP negativo: el déficit externo o la salida de capitales obligó a usar reservas.
        \item 2023 coincide con El Niño, ciclón Yaku y conflictos $\Rightarrow$ menos exportaciones y más presión.
      \end{itemize}
    \end{column}
  \end{columns}
  \vspace{0.7em}
  \begin{block}{Determinantes de la variación}
    \footnotesize El cambio en las reservas responde a dos factores: el \textbf{resultado de la balanza de pagos} del año y el \textbf{efecto valuación} (precio del oro y de las divisas). La suma de ambos explica el $\Delta$ de reservas.
  \end{block}
\end{frame}

\section{Factores de Riesgo}
\begin{frame}{Vulnerabilidades externas}
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \textbf{Choques de oferta y clima}
      \begin{itemize}\footnotesize
        \item \textbf{El Niño:} en 2023 la anchoveta cayó $\sim$51\% y hubo recesión técnica. El BCRP (2026) ya asume un escenario \emph{fuerte}.
        \item Alta concentración en minería: paradas o conflictos golpean la balanza comercial.
        \item Términos de intercambio en su nivel más alto desde 1950 $\Rightarrow$ fortaleza hoy, pero \textbf{reversible}.
      \end{itemize}
    \end{column}
    \begin{column}{0.5\textwidth}
      \textbf{Entorno internacional}
      \begin{itemize}\footnotesize
        \item Fragmentación del comercio y amenazas arancelarias elevan la incertidumbre.
        \item Dependencia de la demanda de \textbf{China} (principal comprador de cobre).
        \item Riesgo de volatilidad financiera y salida de capitales.
        \item Las \textbf{RIN} son la principal línea de defensa.
      \end{itemize}
    \end{column}
  \end{columns}
  \vfill
\end{frame}

\section{Conclusiones}
\begin{frame}{Conclusiones y perspectivas}
  \begin{itemize}
    \item El sector externo peruano pasó de \textbf{déficits recurrentes} a \textbf{superávits récord}, impulsado por la balanza comercial de bienes y los altos precios de los metales.
    \item El mismo factor alimentó el déficit del ingreso primario (utilidades mineras) $\Rightarrow$ carácter \textbf{procíclico} de las cuentas externas.
    \item El giro de la CC se reflejó en la CF y en un fuerte fortalecimiento de las \textbf{RIN} (2024--2025).
    \item BCRP (2026): superávit de CC de 3{,}5\% a 4{,}3\% del PBI; camino a \textbf{cinco años consecutivos} de superávit (2023--2027).
    \item \textbf{Reto:} manejar la dependencia de los metales, mitigar El Niño y sostener colchones fiscales y de reservas.
  \end{itemize}
\end{frame}
